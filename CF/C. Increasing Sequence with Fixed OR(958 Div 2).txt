const int N = 2e5 + 10;

// Adjacency list where each element is a pair of (neighbor, edge index)
vector<pair<int,int>>s[N];
int ans;
 
void dfs(int vertex,int parent,int w=0){//indices of curr vertex and parent
    ans=max(ans,w);
    for(pair<int,int> i: s[vertex]){//traverse all neighbor of cur vertex
        if(i.second!=parent)//no back traversal
        dfs(i.first,i.second,w+(i.second<parent));// w increses if the current edge index is less than the parent's edge index
    }
}
